    69→function GlassShape({
    70→  position,
    71→  scale = 1,
    72→  speed = 0.2,
    73→}: {
    74→  position: [number, number, number]
    75→  scale?: number
    76→  speed?: number
    77→}) {
    78→  const ref = useRef<THREE.Mesh>(null)
    79→  useFrame((state) => {
    80→    if (ref.current) {
    81→      ref.current.rotation.y = state.clock.elapsedTime * speed
    82→      ref.current.rotation.x = state.clock.elapsedTime * speed * 0.5
    83→    }
    84→  })
    85→  return (
    86→    <Float speed={1} rotationIntensity={0.3} floatIntensity={0.8}>
    87→      <Icosahedron ref={ref} args={[1, 0]} position={position} scale={scale}>
    88→        <meshStandardMaterial
    89→          color="#8A2BE2"
    90→          metalness={0.8}
    91→          roughness={0.2}
    92→          wireframe
    93→          transparent
    94→          opacity={0.35}
    95→        />
    96→      </Icosahedron>
    97→    </Float>
    98→  )
    99→}

   162→// ============ CUSTOM CURSOR ============
   163→function CustomCursor() {
   164→  const [position, setPosition] = useState({ x: -100, y: -100 })
   165→  const [isPointer, setIsPointer] = useState(false)
   166→
   167→  useEffect(() => {
   168→    const move = (e: MouseEvent) => {
   169→      // eslint-disable-next-line react-hooks/set-state-in-effect
   170→      setPosition({ x: e.clientX, y: e.clientY })
   171→      const target = e.target as HTMLElement
   172→      // eslint-disable-next-line react-hooks/set-state-in-effect
   173→      setIsPointer(
   174→        target.tagName === 'A' ||
   175→          target.tagName === 'BUTTON' ||
   176→          target.closest('a') !== null ||
   177→          target.closest('button') !== null ||
   178→          target.getAttribute('role') === 'button'
   179→      )
   180→    }
   181→    window.addEventListener('mousemove', move)
   182→    return () => window.removeEventListener('mousemove', move)
   183→  }, [])
   184→
   185→  return (
   186→    <motion.div
   187→      className="fixed top-0 left-0 z-[9999] pointer-events-none hidden md:block"
   188→      animate={{
   189→        x: position.x - (isPointer ? 12 : 4),
   190→        y: position.y - (isPointer ? 12 : 4),
   191→        scale: isPointer ? 3 : 1,
   192→      }}
   193→      transition={{ type: 'spring', stiffness: 500, damping: 30, mass: 0.3 }}
   194→    >
   195→      <div
   196→        className="rounded-full"
   197→        style={{
   198→          width: isPointer ? '24px' : '8px',
   199→          height: isPointer ? '24px' : '8px',
   200→          backgroundColor: '#8A2BE2',
   201→          opacity: 0.7,
   202→          boxShadow: '0 0 12px #8A2BE2',
   203→        }}
   204→      />
   205→    </motion.div>
   206→  )
   207→}

   228→// ============ ANIMATED COUNTER ============
   229→function AnimatedCounter({
   230→  target,
   231→  duration = 2,
   232→}: {
   233→  target: number
   234→  duration?: number
   235→}) {
   236→  const ref = useRef<HTMLSpanElement>(null)
   237→  const inView = useInView(ref, { once: true, margin: '-50px' })
   238→  const [count, setCount] = useState(0)
   239→
   240→  useEffect(() => {
   241→    if (!inView) return
   242→    let startTime: number | null = null
   243→    let raf: number
   244→    const animate = (timestamp: number) => {
   245→      if (startTime === null) startTime = timestamp
   246→      const progress = Math.min((timestamp - startTime) / (duration * 1000), 1)
   247→      const eased = 1 - Math.pow(1 - progress, 3)
   248→      // eslint-disable-next-line react-hooks/set-state-in-effect
   249→      setCount(Math.floor(eased * target))
   250→      if (progress < 1) raf = requestAnimationFrame(animate)
   251→    }
   252→    raf = requestAnimationFrame(animate)
   253→    return () => cancelAnimationFrame(raf)
   254→  }, [inView, target, duration])
   255→
   256→  return <span ref={ref}>{count}</span>
   257→}
