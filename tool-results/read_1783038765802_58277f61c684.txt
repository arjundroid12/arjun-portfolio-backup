    13→  useFrame((state) => { if (ref.current) { ref.current.rotation.y = state.clock.elapsedTime * 0.2 } })

    15→    <Float speed={1} rotationIntensity={0.3} floatIntensity={0.8}>
    16→      <mesh ref={ref} position={position} scale={scale}>
    17→        <icosahedronGeometry args={[1, 0]} />
    18→        <meshStandardMaterial color={color} metalness={0.8} roughness={0.2} wireframe transparent opacity={0.3} />
    19→      </mesh>

    21→  )
    22→}

    87→// ============ CURTAIN TRANSITION ============
    88→function CurtainTransition({ onWarp }: { onWarp: (type: string) => void }) {
    89→  const [active, setActive] = useState(false)
    90→  const [showConfirm, setShowConfirm] = useState(false)
    91→  const lastScrollY = useRef(0)
    92→  const triggered = useRef(false)
    93→  const dismissTimer = useRef<any>(null)
    94→  const atBottomRef = useRef(false)
    95→
    96→  useEffect(() => {
    97→    const handleScroll = () => {
    98→      const scrollY = window.scrollY
    99→      const windowHeight = window.innerHeight
   100→      const docHeight = document.documentElement.scrollHeight
   101→      const atBottom = scrollY + windowHeight >= docHeight - 50
   102→      const scrollingDown = scrollY > lastScrollY.current
   103→      lastScrollY.current = scrollY
   104→      if (atBottom && scrollingDown && !triggered.current) {
   105→        if (!atBottomRef.current) {
   106→          atBottomRef.current = true; setShowConfirm(true); onWarp('confirm')
   107→          dismissTimer.current = setTimeout(() => { setShowConfirm(false); atBottomRef.current = false }, 5000)
   108→        } else {
   109→          clearTimeout(dismissTimer.current); setShowConfirm(false); triggered.current = true; setActive(true); onWarp('warp')
   110→          setTimeout(() => window.scrollTo({ top: 0, behavior: 'smooth' }), 2200)
   111→          setTimeout(() => { setActive(false); triggered.current = false; atBottomRef.current = false }, 4000)
   112→        }
   113→      }
   114→      if (!atBottom && atBottomRef.current && !triggered.current) { clearTimeout(dismissTimer.current); setShowConfirm(false); atBottomRef.current = false }
   115→    }
   116→    window.addEventListener('scroll', handleScroll, { passive: true })
   117→    return () => window.removeEventListener('scroll', handleScroll)
   118→  }, [onWarp])
   119→
   120→  return (
   121→    <>
   122→      <AnimatePresence>
   123→        {showConfirm && !active && (
   124→          <motion.div initial={{ opacity: 0, y: 30 }} animate={{ opacity: 1, y: 0 }} exit={{ opacity: 0, y: 30 }} className="fixed bottom-8 left-1/2 -translate-x-1/2 z-[150] px-6 py-4 bg-[#1a1a1a] border border-[#8A2BE2]/40 rounded text-sm pointer-events-none">
   125→            <span className="text-white">🚀 Scroll once more to return to top</span>
   126→          </motion.div>
   127→        )}
   128→      </AnimatePresence>
   129→      <AnimatePresence>
   130→        {active && (
   131→          <motion.div className="fixed inset-0 z-[200] pointer-events-none" initial={{ opacity: 1 }} exit={{ opacity: 0 }}>
   132→            <motion.div className="absolute inset-0" initial={{ y: '100%' }} animate={{ y: '-100%' }} transition={{ duration: 2, ease: [0.65, 0, 0.35, 1] }} style={{ background: 'linear-gradient(180deg, #121212 0%, #8A2BE2 30%, #9370DB 50%, #8A2BE2 70%, #121212 100%)' }}>
   133→              <div className="flex items-center justify-center h-full">
   134→                <motion.div initial={{ scale: 0 }} animate={{ scale: [0, 1.5, 1, 0] }} transition={{ duration: 2 }} className="text-5xl">✨</motion.div>
   135→              </div>
   136→            </motion.div>
   137→          </motion.div>
   138→        )}
   139→      </AnimatePresence>
   140→    </>
   141→  )
   142→}
